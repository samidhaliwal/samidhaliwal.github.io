61  double ContraBoostInvJigsaw::GetMinimumMass() const {
62  double Minv1 = GetChildState(0).GetMinimumMass();
63  double Minv2 = GetChildState(1).GetMinimumMass();
64  TLorentzVector Pvis1 = m_DependancyStates[0].GetFourVector();
65  TLorentzVector Pvis2 = m_DependancyStates[1].GetFourVector();
66  double Mvis1 = fabs(Pvis1.M());
67  double Mvis2 = fabs(Pvis2.M());
68  double Minv = max(0.,max(Minv1,Minv2));
69  double Mvismin = min(Mvis1,Mvis2);
70  double Mvismax = max(Mvis1,Mvis2);
71 
72  if(Mvismin <= 0.0 && Minv > 0.0) return (Pvis1+Pvis2).M();
73 
74  if(Minv <= Mvismin){
75  return sqrt( (Pvis1+Pvis2).M2() + 4.*(Minv-Mvismin)*(Minv+Mvismax) );
76  } else {
77  return (Pvis1+Pvis2).M()*Minv/Mvismin;
78  }

81  bool ContraBoostInvJigsaw::AnalyzeEvent(){
82  if(!IsSoundMind())
83  return SetSpirit(false);
84 
85  CalcCoef();
86  double c1 = GetC1();
87  double c2 = GetC2();
88 
89 
90  TLorentzVector Pvis1 = m_DependancyStates[0].GetFourVector();
91  TLorentzVector Pvis2 = m_DependancyStates[1].GetFourVector();
92  TLorentzVector INV = GetParentState().GetFourVector();
93 
94  // go to the rest frame of (Pvis1+Pvis2+INV system)
95  TVector3 Boost = (Pvis1+Pvis2+INV).BoostVector();
96  Pvis1.Boost(-Boost);
97  Pvis2.Boost(-Boost);
98  INV.Boost(-Boost);
99 
100  double E1 = Pvis1.E();
101  double E2 = Pvis2.E();
102  TVector3 P1 = Pvis1.Vect();
103  TVector3 P2 = Pvis2.Vect();
104 
105  double N = (E1+E2 + sqrt( (E1+E2)*(E1+E2) - (Pvis1+Pvis2).M2() + INV.M2() ))/(c1*E1+c2*E2)/2.;
106 
107  c1 *= N;
108  c2 *= N;
109 
110  TLorentzVector INV1,INV2;
111  double Einv1 = (c1-1.)*E1 + c2*E2;
112  double Einv2 = c1*E1 + (c2-1.)*E2;
113  TVector3 Pinv1 = (c1-1.)*P1 - c2*P2;
114  TVector3 Pinv2 = (c2-1.)*P2 - c1*P1;
115 
116  INV1.SetPxPyPzE(Pinv1.X(),Pinv1.Y(),Pinv1.Z(),Einv1);
117  INV2.SetPxPyPzE(Pinv2.X(),Pinv2.Y(),Pinv2.Z(),Einv2);
118  INV1.Boost(Boost);
119  INV2.Boost(Boost);
120 
121  GetChildState(0).SetFourVector(INV1);
122  GetChildState(1).SetFourVector(INV2);
123 
124  return SetSpirit(true);

127  void ContraBoostInvJigsaw::CalcCoef(){
128  double Minv1 = GetChildState(0).GetMinimumMass();
129  double Minv2 = GetChildState(1).GetMinimumMass();
130  TLorentzVector Pvis1 = m_DependancyStates[0].GetFourVector();
131  TLorentzVector Pvis2 = m_DependancyStates[1].GetFourVector();
132  double m1 = max(0.,Pvis1.M());
133  double m2 = max(0.,Pvis2.M());
134  double Minv = max(Minv1,Minv2);
135  double Mvis = min(m1,m2);
136 
137  if(Minv >= Mvis){
138  m_C1 = 1.;
139  m_C2 = 1.;
140  return;
141  }
142  // go to the rest frame of the vis+inv system
143  TLorentzVector INV = GetParentState().GetFourVector();
144  TVector3 Boost = (Pvis1+Pvis2+INV).BoostVector();
145  Pvis1.Boost(-Boost);
146  Pvis2.Boost(-Boost);
147 
148  double MC2 = 2.*( Pvis1.E()*Pvis2.E() + Pvis1.Vect().Dot(Pvis2.Vect()) );
149  double k1 = (m1+m2)*(m1-m2)*(1.-Minv/Mvis) + MC2-2*m1*m2 + (m1+m2)*fabs(m1-m2)*Minv/Mvis;
150  double k2 = -(m1+m2)*(m1-m2)*(1.-Minv/Mvis) + MC2-2*m1*m2 + (m1+m2)*fabs(m1-m2)*Minv/Mvis;
151  double Xbar = sqrt( (k1+k2)*(k1+k2)*(MC2*MC2-4*m1*m1*m2*m2) + 16.*Minv*Minv*(k1*k1*m1*m1 + k2*k2*m2*m2 + k1*k2*MC2) );
152  double N = ( fabs(k1*m1*m1-k2*m2*m2) - 0.5*fabs(k2-k1)*MC2 + 0.5*Xbar )/(k1*k1*m1*m1 + k2*k2*m2*m2 + k1*k2*MC2);
153  m_C1 = 0.5*(1.+N*k1);
154  m_C2 = 0.5*(1.+N*k2);
155  N = sqrt(m_C1*m_C2);
156  m_C1 /= N;
157  m_C2 /= N;
158  }
