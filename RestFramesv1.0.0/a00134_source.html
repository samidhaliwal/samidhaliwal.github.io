194  void TreePlot::ConvertNodeCoordinates(vector<TreePlotNode*>& nodes){
195  double xmin = 0.;
196  double xmax = 1.;
197  double ymin = 0.;
198  double ymax = 1.;
199  if(m_Type == kFrameTree) ymax = 0.8;
200  int Nnode = nodes.size();
201  for(int i = 0; i < Nnode; i++){
202  double new_x = nodes[i]->GetX();
203  double new_y = nodes[i]->GetY();
204  new_x = xmin + (xmax-xmin)*(new_x+0.5)/double(m_Ncol[int(new_y)]);
205  new_y = ymin + (ymax-ymin)*(1.-(new_y+0.5)/double(m_Nrow));
206  nodes[i]->SetX(new_x);
207  nodes[i]->SetY(new_y);
208  }

282  expand_child = true;

291  child_nodePtr->SetY(irow+1);
292  child_nodePtr->SetFrame(child);

296  m_TreeLinks.push_back(linkPtr);
297  }
298  m_Ncol[irow+1]++;
299  AddFrame(child);
300  } else {
301  TreePlotNode* child_nodePtr = new TreePlotNode();
302  child_nodePtr->SetX(m_Ncol[irow+1]);

362  void TreePlot::FillJigsawLink(const Jigsaw& jigsaw){
363  int Nsplit = jigsaw.GetNChildren();
364  if(!jigsaw.GetGroup()) return;
365  TreePlotNode* high_old = nullptr;
366  TreePlotNode* high_new = nullptr;
367  for(int s = 0; s < Nsplit; s++){
368  RFList<RestFrame> frames = jigsaw.GetChildFrames(s)+
369  jigsaw.GetDependancyFrames(s);
370  int Nnode = m_TreeNodes.size();
371  TreePlotNode* last_nodePtr = nullptr;
372  double high = -1.;
373  for(int n = 0; n < Nnode; n++){
374  TreePlotNode* nodePtr = m_TreeNodes[n];
375  const RestFrame& frame = nodePtr->GetFrame();
376  if(frames.Contains(frame)){
377  nodePtr->AddJigsaw(jigsaw);
378  if(jigsaw.GetGroup().ContainsFrame(frame)){
379  if(nodePtr->GetY() > high){
380  high_new = nodePtr;
381  high = nodePtr->GetY();
382  }
383  }
384  if(last_nodePtr){
385  if(!(last_nodePtr->GetFrame() == nodePtr->GetFrame())){
386  TreePlotLink* linkPtr = new TreePlotLink(last_nodePtr,nodePtr);
387  linkPtr->SetJigsaw(jigsaw);
388  m_LeafLinks.push_back(linkPtr);
389  }
390  }
391  last_nodePtr = nodePtr;
392  }
393  }
394 
395  if(s != 0){
396  TreePlotLink* linkPtr = new TreePlotLink(high_old,high_new);
397  linkPtr->SetWavy(true);
398  linkPtr->SetJigsaw(jigsaw);
399  m_LeafLinks.push_back(linkPtr);
400  }
401  high_old = high_new;
402  }
403  }

423  void TreePlot::DrawLink(TreePlotLink* linkPtr){
424  double x0 = linkPtr->GetNode1()->GetX();
425  double y0 = linkPtr->GetNode1()->GetY();
426  double x1 = linkPtr->GetNode2()->GetX();
427  double y1 = linkPtr->GetNode2()->GetY();
428 
429  int icolor = color_Default;
430  int icolor_fill = -1;
431  int istyle = style_Default;
432  int isize = int(m_Node_R*70.);
433  bool DoWavy = linkPtr->DoWavy();
434 
435  const Jigsaw& jigsaw = linkPtr->GetJigsaw();
436 
437  if(!jigsaw.IsEmpty()){
438  if(linkPtr->DoLabel()){
439  int priority = GetJigsawPriority(jigsaw.GetNChildren(), jigsaw.GetNDependancyStates());
440  icolor = color_Node[priority];
441  icolor_fill = color_fill_Node[priority];
442  } else {
443  int Nj = m_Jigsaws.GetN();
444  int index = m_Jigsaws.GetIndex(jigsaw);
445  x0 += (double(index+1)-double(Nj+1)/2.)*m_Node_R*1./max(4.,double(Nj));
446  x1 += (double(index+1)-double(Nj+1)/2.)*m_Node_R*1./max(4.,double(Nj));
447  icolor = m_JigsawColorMap[&jigsaw];
448  if(DoWavy){
449  istyle = style_Leaf;
450  isize++;
451  }
452  }
453  }
454 
455  if(fabs(y0-y1) > 1e-10){ // nodes are at different heights - draw line
456  TLine *line = new TLine(x0,y0,x1,y1);
457  line->SetLineColor(icolor);
458  line->SetLineWidth(isize);
459  line->SetLineStyle(istyle);
460  m_CanvasPtr->cd();
461  line->Draw();
462  AddTObject(line);
463  } else { // nodes are at same height - draw arc
464  TArc *arc;
465  double c = fabs(x0-x1);
466  double h = 1./double(2*m_Nrow)*0.8;
467  if(h > c/2.){
468  double R = c/2.;
469  if(h > R+m_Node_R) arc = new TArc((x0+x1)/2.,y0+m_Node_R,R, 0., 180.);
470  else arc = new TArc((x0+x1)/2.,y0+h-R,R, 0., 180.);
471  } else {
472  double R = h/2. + c*c/(8.*h);
473  arc = new TArc((x0+x1)/2.,y0+h-R,R,
474  -asin(c/(2.*R))*180/TMath::Pi()+90.,
475  asin(c/(2.*R))*180/TMath::Pi()+90.);
476  }
477  arc->SetLineColor(icolor);
478  arc->SetLineStyle(istyle);
479  arc->SetFillStyle(0);
480  arc->SetLineWidth(isize);
481  m_CanvasPtr->cd();
482  arc->Draw("only");
483  AddTObject(arc);
484  }
485 
486  if(linkPtr->DoLabel()){
487  double x = x0;
488  double y = (y0+y1)/2.;
489 
490  char* slabel = new char[500];
491  sprintf(slabel,"#bf{%s}",linkPtr->GetLabel().c_str());
492  TLatex* lat = new TLatex(x,y,slabel);
493  lat->SetTextAlign(22);
494  lat->SetNDC();
495  lat->SetTextSize(1.);
496  lat->SetTextFont(132);
497  double Xsize = lat->GetXsize();
498  double Ysize = lat->GetYsize();
499  double scale = max( Xsize/(6.*m_Node_R), 1.25*Ysize/(fabs(y0-y1)-2.*m_Node_R) );
500  lat->SetTextSize(1./scale);
501  lat->SetTextColor(icolor);
502  lat->SetLineColor(icolor);
503 
504  Xsize = lat->GetXsize()*1.03;
505  Ysize = lat->GetYsize();
506  TBox *border = new TBox(x-Xsize/2., y-Ysize/2., x+Xsize/2.,y+Ysize/2.);
507  border->SetLineColor(icolor);
508  border->SetLineWidth(2);
509  border->SetLineStyle(1);
510  border->SetFillColor(icolor_fill);
511  m_CanvasPtr->cd();
512  border->Draw("l");
513  m_CanvasPtr->cd();
514  lat->DrawLatex(x,y,slabel);
515  delete[] slabel;
516  AddTObject(lat);
517  AddTObject(border);
518  }
519  }

530

532 
533  if(!frame.IsEmpty()){
534  icolor = m_FrameColorMap[frame.GetType()];
535  icolor_fill = m_FrameColorFillMap[frame.GetType()];
536  }
537

540  int Njigsaw = jigsaws.GetN();

542  double R = 1.03 + double(Njigsaw-i)*0.08;
543  const Jigsaw& jigsaw = jigsaws[i];

545  TBox* ring = new TBox(x-m_Node_R*R*0.88,y-m_Node_R*R*0.88,

556  ring->SetFillColor(m_JigsawColorMap[&jigsaw]);
557 
558  m_CanvasPtr->cd();
559  ring->Draw();
560  AddTObject(ring);
561  }
562  }
563  }
564  if(square){
565  TBox *box = new TBox(x-m_Node_R*0.88,y-m_Node_R*0.88,
566  x+m_Node_R*0.88,y+m_Node_R*0.88);
567  box->SetLineColor(icolor);
568  box->SetFillColor(icolor_fill);
569  box->SetLineStyle(istyle);
570  box->SetLineWidth(iwidth);
571  m_CanvasPtr->cd();
572  box->Draw("l");
573  AddTObject(box);
574  } else {
575  TArc *circle = new TArc(x,y,m_Node_R);
576  circle->SetLineColor(icolor);
577  circle->SetFillColor(icolor_fill);
578  circle->SetLineStyle(istyle);
579  circle->SetLineWidth(iwidth);
580  m_CanvasPtr->cd();
581  circle->Draw();
582  AddTObject(circle);
583  }
584 
585  if(nodePtr->DoLabel()){

603  string TreePlot::GetStateTitle(const State& state){
604  RFList<RestFrame> frames = state.GetListFrames();
605  int Nf = frames.GetN();
606  string title = "";
607  if(Nf > 2) title.append("#splitline{");
608  title.append(frames.Get(0).GetTitle());
609  for(int f = 1; f < Nf; f++){
610  if(f%((Nf+1)/2) == 0 && Nf > 2) title.append("}{");
611  title.append("+ ");
612  title.append(frames.Get(f).GetTitle());
613  }
614  if(Nf > 2) title.append("}");
