118  bool CombinatoricJigsaw::InitializeJigsawExecutionList(RFList<Jigsaw>& exec_jigsaws){
119  if(!IsSoundMind()) return false;
120  if(exec_jigsaws.Contains(*this)) return true;
121 
122  m_ExecuteJigsaws.Clear();
123 
124  // Add group dependancy jigsaws first
125  RFList<Jigsaw> group_jigsaws;
126  FillGroupJigsawDependancies(group_jigsaws);
127  group_jigsaws -= *this;
128 
129  int Ngroup = group_jigsaws.GetN();
130  for(int i = Ngroup-1; i >= 0; i--){
131  Jigsaw& jigsaw = group_jigsaws[i];
132  m_DependancyJigsaws -= jigsaw;
133  if(!exec_jigsaws.Contains(jigsaw))
134  if(!jigsaw.InitializeJigsawExecutionList(exec_jigsaws))
135  return SetMind(false);
136  }
137  // Satisfy dependancy jigsaws
138  while(m_DependancyJigsaws.GetN() > 0){
139  Jigsaw& jigsaw = m_DependancyJigsaws[m_DependancyJigsaws.GetN()-1];
140  if(exec_jigsaws.Contains(jigsaw)){
141  m_DependancyJigsaws -= jigsaw;
142  continue;
143  }
144  if(!jigsaw.DependsOnJigsaw(*this)){
145  if(!jigsaw.InitializeJigsawExecutionList(exec_jigsaws)){
146  return SetMind(false);
147  }
148  m_DependancyJigsaws -= jigsaw;
149  continue;
150  }
151  RFList<Jigsaw> temp_exec_jigsaws = exec_jigsaws;
152  temp_exec_jigsaws += m_ExecuteJigsaws;
153  temp_exec_jigsaws += *this;
154  if(!jigsaw.InitializeJigsawExecutionList(temp_exec_jigsaws))
155  return SetMind(false);
156  temp_exec_jigsaws -= *this;
157  temp_exec_jigsaws -= exec_jigsaws;
158 
159  m_DependancyJigsaws -= temp_exec_jigsaws;
160  m_ExecuteJigsaws += temp_exec_jigsaws;
161  }
162 
163  exec_jigsaws += *this;
164  exec_jigsaws += m_ExecuteJigsaws;
165  return true;
166  }

202  bool CombinatoricJigsaw::InitializeCombinatoric(){
203  if(!IsSoundMind())
204  return SetSpirit(false);
205 
206  if(!GetGroup())
207  return SetSpirit(false);
208 
209  if(!GetParentState())
210  return SetSpirit(false);
211 
212  m_InputStates.Clear();
213  m_InputStates = GetParentState().GetElements();
214 
215  m_NForChild.clear();
216  m_NExclusive.clear();
217 
218  int NTOTOT = 0;
219  bool exclTOTOT = true;
220  int Nc = GetNChildren();
221  for(int i = 0; i < Nc; i++){
222  RFList<RestFrame> frames = GetChildState(i).GetListFrames();
223  int Nf = frames.GetN();
224  int NTOT = 0;
225  bool exclTOT = true;
226  for(int f = 0; f < Nf; f++){
227  int N = -1;
228  bool excl = false;
229  GetGroup().GetNElementsForFrame(frames[f], N, excl);
230  if(N < 0) return SetSpirit(false);
231  NTOT += N;
232  exclTOT = exclTOT && excl;
233  }
234  m_NForChild[&GetChildState(i)] = NTOT;
235  m_NExclusive[&GetChildState(i)] = exclTOT;
236  NTOTOT += NTOT;
237  exclTOTOT = exclTOTOT && exclTOT;
238  }
239 
240  if(m_InputStates.GetN() < NTOTOT){
241  m_Log << LogWarning;
242  m_Log << "Unable to execute Jigsaw. ";
243  m_Log << "Insufficienct number of inputs.";
244  m_Log << m_End;
245  return SetSpirit(false);
246  }
247 
248  if(exclTOTOT &&
249  (m_InputStates.GetN() != NTOTOT)){
250  m_Log << LogWarning;
251  m_Log << "Unable to execute Jigsaw. ";
252  m_Log << "Incorrect number of exclusive inputs: ";
253  m_Log << NTOTOT << " (required) != ";
254  m_Log << m_InputStates.GetN() << " (provided)";
255  m_Log << m_End;
256  return SetSpirit(false);
257  }
258 
259  return SetSpirit(true);

262  bool CombinatoricJigsaw::LoopCombinatoric(){
263  int Ninput = m_InputStates.GetN();
264  int Nout = GetNChildren();
265 
266  int N_comb = 1;
267  for(int i = 0; i < Ninput; i++) N_comb *= Nout;
268 
269  vector<int> Nhem;
270  for(int i = 0; i < Nout; i++)
271  Nhem.push_back(0);
272 
273  int c_max = -1;
274  double metric_max = -1;
275 
276  for(int c = 0; c < N_comb; c++){
277  int key = c;
278  for(int i = 0; i < Nout; i++){
279  Nhem[i] = 0;
280  GetChildState(i).ClearElements();
281  }
282 
283  // set output states for combinatoric;
284  for(int i = 0; i < Ninput; i++){
285  int ihem = key%Nout;
286  key /= Nout;
287  Nhem[ihem]++;
288  GetChildState(ihem).AddElement(m_InputStates[i]);
289  }
290 
291  // check validity of combinatoric
292  bool valid = true;
293  for(int i = 0; i < Nout; i++){
294  if(m_NExclusive[&GetChildState(i)]){
295  if(Nhem[i] != m_NForChild[&GetChildState(i)])
296  valid = false;
297  } else {
298  if(Nhem[i] < m_NForChild[&GetChildState(i)])
299  valid = false;
300  }
301  }
302  if(!valid)
303  continue;
304 
305  // Execute depedancy Jigsawsfor this combintoric
306  ExecuteDependancyJigsaws();
307 
308  // Evaluate metric for this combinatoric
309  double metric = EvaluateMetric();
310 
311  if(metric >= metric_max){
312  metric_max = metric;
313  c_max = c;
314  }
315  }
316 
317  if(c_max < 0){
318  m_Log << LogWarning;
319  m_Log << "Unable to find combinatoric with positive metric";
320  m_Log << m_End;
321  return SetSpirit(false);
322  }
323 
324  // Set outputs to best combinatoric
325  for(int i = 0; i < 2; i++)
326  GetChildState(i).ClearElements();
327  int key = c_max;
328  for(int i = 0; i < Ninput; i++){
329  int ihem = key%Nout;
330  key /= Nout;
331  GetChildState(ihem).AddElement(m_InputStates[i]);
332  }
333 
334  // Execute depedancy Jigsaws
335  ExecuteDependancyJigsaws();
336 
337  return SetSpirit(true);
338  }
