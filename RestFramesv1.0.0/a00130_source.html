105  void SelfAssemblingRecoFrame::Assemble(){
106  if(m_IsAssembled) Disassemble();
107  if(!IsSoundMind()){
108  UnSoundMind(RF_FUNCTION);
109  return;
110  }
111 
112  // new Visible States
113  RFList<VisibleState> states;
114  // new Frames associated with States
115  vector<RestFrame*> frames;
116  // States' four-vector
117  vector<TLorentzVector> Ps;
118 
119  // clear unassembled lists
120  m_ChildFrames_UnAssembled.Clear();
121  m_ChildFrames_UnAssembled += GetChildren();
122  int N = GetNChildren();
123  for(int i = 0; i < N; i++){
124  ReconstructionFrame& frame = GetChildFrame(i);
125  bool expand_frame = false;
126  if(m_ChildStates[&frame].GetN() == 1)
127  if(m_ChildStates[&frame][0].IsCombinatoricState()){
128  expand_frame = true;
129  RFList<VisibleState> elements =
130  static_cast<CombinatoricState&>(m_ChildStates[&frame][0]).GetElements();
131  int Nelement = elements.GetN();
132  for(int e = 0; e < Nelement; e++){
133  VisibleState& element = elements[e];
134  RestFrame& new_frame = GetNewVisibleFrame(frame.GetName(),frame.GetTitle());
135  element.AddFrame(new_frame);
136  frames.push_back(&new_frame);
137  TLorentzVector V = element.GetFourVector();
138  if(V.M() < 0.) V.SetVectM(V.Vect(),0.);
139  Ps.push_back(V);
140  states.Add(element);
141  }
142  }
143  if(!expand_frame){
144  TLorentzVector V = m_ChildStates[&frame].GetFourVector();
145  if(V.M() < 0.) V.SetVectM(V.Vect(),0.);
146  Ps.push_back(V);
147  frames.push_back(&frame);
148  }
149  }
150  RemoveChildren();
151  m_ChildStates.clear();
152  AssembleRecursive(*this, frames, Ps);
153 
154  if(!InitializeTreeRecursive()){
155  m_Log << LogWarning;
156  m_Log << "Problem with recursive tree after assembly";
157  m_Log << m_End;
158  SetBody(false);
159  return;
160  }
161  SetMind(true);
162  const LabRecoFrame& lab_frame = static_cast<const LabRecoFrame&>(GetLabFrame());
163  lab_frame.AddTreeStates(states);
164  if(!InitializeAnalysisRecursive()){
165  m_Log << LogWarning;
166  m_Log << "Problem connecting states after assembly";
167  m_Log << m_End;
168  SetMind(false);
169  return;
170  }
171  lab_frame.RemoveTreeStates(states);
172 
173  m_IsAssembled = true;
174  }

176  void SelfAssemblingRecoFrame::AssembleRecursive(RestFrame& frame, vector<RestFrame*>& frames, vector<TLorentzVector>& Ps){
177 
178  int Ninput = frames.size();
179  if(Ninput <= 1){
180  for(int i = 0; i < Ninput; i++) frame.AddChildFrame(*frames[i]);
181  return;
182  }
183 
184  TLorentzVector TOT(0.,0.,0.,0.);
185  for(int i = 0; i < Ninput; i++) TOT += Ps[i];
186  TVector3 boost = TOT.BoostVector();
187  for(int i = 0; i < Ninput; i++){
188  Ps[i].Boost(-boost);
189  }
190 
191  int ip_max[2];
192  int jp_max[2];
193  for(int i = 0; i < 2; i++) ip_max[i] = -1;
194  for(int i = 0; i < 2; i++) jp_max[i] = -1;
195  double val_max = -1.;
196  // Loop over all 2-jet seed probes
197  int ip[2], jp[2];
198  for(ip[0] = 0; ip[0] < Ninput-1; ip[0]++){
199  for(ip[1] = ip[0]+1; ip[1] < Ninput; ip[1]++){
200  TVector3 nRef = Ps[ip[0]].Vect().Cross(Ps[ip[1]].Vect());
201  int Nhem[2];
202  TLorentzVector hem[2];
203  for(int i = 0; i < 2; i++){
204  Nhem[i] = 0;
205  hem[i].SetPxPyPzE(0.,0.,0.,0.);
206  }
207  // Loop over all jets
208  for(int i = 0; i < Ninput; i++){
209  if((i == ip[0]) || (i ==ip[1])) continue;
210  int ihem = int(Ps[i].Vect().Dot(nRef) > 0.);
211  Nhem[ihem]++;
212  hem[ihem] += Ps[i];
213  }
214  // assign 2 probes
215  for(jp[0] = 0; jp[0] < 2; jp[0]++){
216  for(jp[1] = 0; jp[1] < 2; jp[1]++){
217  if(jp[0] == jp[1] && Nhem[(jp[0]+1)%2] == 0) continue;
218  TLorentzVector hem_probes[2];
219  for(int i = 0; i < 2; i++) hem_probes[i] = hem[i];
220  for(int i = 0; i < 2; i++) hem_probes[jp[i]] += Ps[ip[i]];
221  double val = hem_probes[0].P() + hem_probes[1].P();
222  if(val > val_max){
223  val_max = val;
224  for(int i = 0; i < 2; i++) ip_max[i] = ip[i];
225  for(int i = 0; i < 2; i++) jp_max[i] = jp[i];
226  }
227  }
228  }
229  }
230  }
231  vector<RestFrame*> child_frames[2];
232  vector<TLorentzVector> child_Ps[2];
233  TLorentzVector hem[2];
234  for(int i = 0; i < 2; i++){
235  hem[i].SetPxPyPzE(0.,0.,0.,0.);
236  }
237  for(int i = 0; i < 2; i++){
238  child_frames[jp_max[i]].push_back(frames[ip_max[i]]);
239  child_Ps[jp_max[i]].push_back(Ps[ip_max[i]]);
240  hem[jp_max[i]] += Ps[ip_max[i]];
241  }
242  TVector3 nRef = Ps[ip_max[0]].Vect().Cross(Ps[ip_max[1]].Vect());
243  for(int i = 0; i < Ninput; i++){
244  if((i == ip_max[0]) || (i == ip_max[1])) continue;
245  int ihem = int(Ps[i].Vect().Dot(nRef) > 0.);
246  child_frames[ihem].push_back(frames[i]);
247  child_Ps[ihem].push_back(Ps[i]);
248  hem[ihem] += Ps[i];
249  }
250 
251  int flip = int(hem[1].M() > hem[0].M());
252  for(int i = 0; i < 2; i++){
253  int j = (i+flip)%2;
254  if(child_frames[j].size() == 1){
255  frame.AddChildFrame(*child_frames[j][0]);
256  } else {
257  RestFrame& new_frame = GetNewDecayFrame(GetName(),GetTitle());
258  frame.AddChildFrame(new_frame);
259  AssembleRecursive(new_frame, child_frames[j], child_Ps[j]);
260  }
261  }
262  }

264  bool SelfAssemblingRecoFrame::AnalyzeEventRecursive(){
265  // Disassemble Frame tree if it assembled
266  if(m_IsAssembled) Disassemble();
267  if(!ReconstructionFrame::AnalyzeEventRecursive()){
268  m_Log << LogWarning;
269  m_Log << "Unable to recursively analyze event with ";
270  m_Log << "disassembled SelfAssemblingRecoFrame" << m_End;
271  return SetSpirit(false);
272  }
273  // Assemble Frame tree
274  Assemble();
275  if(!ReconstructionFrame::AnalyzeEventRecursive()){
276  m_Log << LogWarning;
277  m_Log << "Unable to recursively analyze event with ";

279  return SetSpirit(false);
280  }
281  return SetSpirit(true);
282  }
