36 void example_02_InvisibleJigsaws_Wtolnu(string output_name = "output_02.root"){
37 
38  double mW = 81.;
39  int Ngen = 100000;
40 
41  // Set up toy generation and event analysis trees:
42  LabGenFrame LAB_G("LAB_G","LAB");
43  DecayGenFrame W_G("W_G","W");
44  VisibleGenFrame L_G("L_G","#it{l}");
45  InvisibleGenFrame NU_G("NU_G","#nu");
46 
47  LAB_G.SetChildFrame(W_G);
48  W_G.AddChildFrame(L_G);
49  W_G.AddChildFrame(NU_G);
50 
51  if(!LAB_G.InitializeTree()) cout << "Problem with generator tree" << endl;
52 
53  W_G.SetMass(mW);
54 
55  if(!LAB_G.InitializeAnalysis()) cout << "Problem with generator tree" << endl;
56 
57  LabRecoFrame LAB_R("LAB_R","LAB");
58  DecayRecoFrame W_R("W_R","W");
59  VisibleRecoFrame L_R("L_R","#it{l}");
60  InvisibleRecoFrame NU_R("NU_R","#nu");
61 
62  LAB_R.SetChildFrame(W_R);
63  W_R.AddChildFrame(L_R);
64  W_R.AddChildFrame(NU_R);
65 
66  if(!LAB_R.InitializeTree()) cout << "Problem with reconstruction tree" << endl;
67 
68  TreePlot* tree_plot = new TreePlot("TreePlot","TreePlot");
69 
70  // generator tree
71  tree_plot->SetFrameTree(LAB_G);
72  tree_plot->Draw("GenTree", "Generator Tree");
73 
74  // reco tree
75  tree_plot->SetFrameTree(LAB_R);
76  tree_plot->Draw("RecoTree", "Reconstruction Tree");
77 
78 
79  // Now we add invisible jigsaws
80  InvisibleGroup INV("INV","Neutrino Jigsaws");
81  INV.AddFrame(NU_R);
82 
83  SetMassInvJigsaw MassJigsaw("MASS_JIGSAW","mass Jigsaw");
84  INV.AddJigsaw(MassJigsaw);
85 
86  SetRapidityInvJigsaw RapidityJigsaw("RAPIDITY_JIGSAW","rapidity Jigsaw");
87  INV.AddJigsaw(RapidityJigsaw);
88  RapidityJigsaw.AddVisibleFrame(L_R);
89 
90  if(!LAB_R.InitializeAnalysis()) cout << "Problem with jigsaws" << endl;
91 
92  // Invisible Jigsaws
93  tree_plot->SetGroupTree(INV);
94  tree_plot->Draw("InvTree", "InvisibleJigsaws");
95 
96  // Declare observables for histogram booking
97  HistPlot* hist_plot = new HistPlot("HistPlot","W #rightarrow #it{l} #nu");
98 
99  const HistPlotVar& MW = hist_plot->GetNewVar("MW", "M_{W} [GeV]", 0., 90.);
100  const HistPlotVar& pTW = hist_plot->GetNewVar("pTW","p_{T}^{W} / m_{W}",0.,1.);
101  const HistPlotVar& cosW = hist_plot->GetNewVar("cosW","cos #theta_{W}", -1., 1.);
102  const HistPlotVar& dphiW = hist_plot->GetNewVar("dphiW", "#Delta #phi_{W}", 0., 2.*acos(-1.));
103  const HistPlotVar& DcosW = hist_plot->GetNewVar("DcosW","#theta_{W} - #theta_{W}^{true}", -1., 1.);
104  const HistPlotVar& DdphiW = hist_plot->GetNewVar("DdphiW","#Delta #phi_{W} - #Delta #phi_{W}^{true}", -1., 1.);
105 
106  hist_plot->AddHist(MW);
107  hist_plot->AddHist(cosW);
108  hist_plot->AddHist(dphiW);
109  hist_plot->AddHist(DcosW);
110  hist_plot->AddHist(DdphiW);
111  hist_plot->AddHist(MW, cosW);
112  hist_plot->AddHist(MW, pTW);
113  hist_plot->AddHist(cosW, pTW);
114  hist_plot->AddHist(dphiW, pTW);
115  hist_plot->AddHist(DcosW, pTW);
116  hist_plot->AddHist(DdphiW, pTW);
117 
118  for(int igen = 0; igen < Ngen; igen++){
119  if(igen%(Ngen/10) == 0) cout << "Generating event " << igen << " of " << Ngen << endl;
120 
121  // generate event
122  LAB_G.ClearEvent(); // clear the gen tree
123  double PTW = mW*gRandom->Rndm();
124  LAB_G.SetTransverseMomenta(PTW); // give the W some Pt
125  double PzW = mW*(2.*gRandom->Rndm()-1.);
126  LAB_G.SetLongitudinalMomenta(PzW); // give the W some Pz
127  LAB_G.AnalyzeEvent(); // generate a new event
128 
129  // analyze event
130  LAB_R.ClearEvent(); // clear the reco tree
131  L_R.SetLabFrameFourVector(L_G.GetFourVector()); // Set lepton 4-vec
132  TVector3 MET = LAB_G.GetInvisibleMomentum(); // Get the MET from gen tree
133  MET.SetZ(0.);
134  INV.SetLabFrameThreeVector(MET); // Set the MET in reco tree
135  LAB_R.AnalyzeEvent(); // analyze the event
136 
137  // calculate observables
138  MW = W_R.GetMass();
139  pTW = PTW/mW;
140  cosW = W_R.GetCosDecayAngle();
141  dphiW = LAB_R.GetDeltaPhiDecayPlanes(W_R);
142  double cosWgen = W_G.GetCosDecayAngle();
143  double dphiWgen = LAB_G.GetDeltaPhiDecayPlanes(W_G);
144  DcosW = asin(sqrt(1.-cosW*cosW)*cosWgen-sqrt(1.-cosWgen*cosWgen)*cosW);
145  DdphiW = asin(sin(dphiW-dphiWgen));
146 
147  hist_plot->Fill();
148  }
149 
150  hist_plot->Draw();
151  hist_plot->WriteOutput(output_name);
152  tree_plot->WriteOutput(output_name);
153 
154 }
