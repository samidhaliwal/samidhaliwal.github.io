353  bool RestFrame::FindPathToFrame(const RestFrame& dest_frame, const RestFrame& prev_frame,
354  vector<TVector3>& boosts) const {
355  if(IsSame(dest_frame)) return true;
356 
357  vector<const RestFrame*> try_frames;
358  vector<TVector3> try_boosts;
359 
360  if(!GetParentFrame().IsEmpty()){
361  try_frames.push_back(&GetParentFrame());
362  try_boosts.push_back(GetParentBoostVector());
363  }
364  int Nchild = GetNChildren();
365  for(int i = 0; i < Nchild; i++){
366  try_frames.push_back(&GetChildFrame(i));
367  try_boosts.push_back(GetChildBoostVector(GetChildFrame(i)));
368  }
369 
370  int Ntry = try_frames.size();
371  for(int i = 0; i < Ntry; i++){
372  const RestFrame* nextPtr = try_frames[i];
373  if(nextPtr->IsSame(prev_frame)) continue;
374  boosts.push_back(try_boosts[i]);
375  if(nextPtr->FindPathToFrame(dest_frame,*this,boosts))
376  return true;
377  boosts.pop_back();
378  }
379  return false;
380  }

539  double RestFrame::GetVisibleShape() const {
540  if(!IsSoundSpirit()){
541  UnSoundSpirit(RF_FUNCTION);
542  return 0.;
543  }
544  double Psum = 0.;
545  double Xsum = 0.;
546  int N = GetNChildren();

551  for(int j = i+1; j < N; j++){

685  double RestFrame::GetDeltaPhiDecayAngle(const TVector3& axis, const RestFrame& frame) const {
686  if(!IsSoundSpirit()){
687  UnSoundSpirit(RF_FUNCTION);
688  return 0.;
689  }
690 
691  if(IsLabFrame())
692  return 0.;
693  TLorentzVector Pthis = GetFourVector(frame);
694  TLorentzVector Pparent = GetParentFrame().GetFourVector(frame);
695  TLorentzVector Pchild = GetChildFrame(0).GetFourVector(frame);
696 
697  TVector3 boost_par = Pthis.BoostVector();
698  boost_par = boost_par.Dot(axis.Unit())*axis.Unit();
699  Pthis.Boost(-boost_par);
700  Pparent.Boost(-boost_par);
701  Pchild.Boost(-boost_par);
702  TVector3 boost_perp = Pthis.BoostVector();
703  Pparent.Boost(-boost_perp);
704  Pchild.Boost(-boost_perp);
705 
706  TVector3 V1 = Pparent.Vect();
707  TVector3 V2 = Pchild.Vect();
708  V1 = V1 - V1.Dot(axis.Unit())*axis.Unit();
709  V2 = V2 - V2.Dot(axis.Unit())*axis.Unit();
710  return V1.Angle(V2);
