54  bool MinMassesCombJigsaw::AnalyzeEvent(){
55  if(!IsSoundMind() || !GetGroup())
56  return SetSpirit(false);
57 
58  if(!InitializeCombinatoric()){
59  m_Log << LogWarning;
60  m_Log << "Problem initializing event info" << m_End;
61  return SetSpirit(false);
62  }
63 
64  int Ninput = m_InputStates.GetN();
65  int Nout = GetNChildren();
66  int Ndeps = m_DependancyStates.size();
67 
68  bool DO_N3 = (Nout == 2) && (Ndeps <= 0);
69  if(DO_N3){
70  DO_N3 = (m_NForChild[&GetChildState(0)] == 1) &&
71  (m_NForChild[&GetChildState(1)] == 1) &&
72  !m_NExclusive[&GetChildState(0)] &&
73  !m_NExclusive[&GetChildState(1)];
74  }
75  if(!DO_N3)
76  return CombinatoricJigsaw::LoopCombinatoric();
77 
78  // DO 4 N^3 calculation
79  vector<TLorentzVector> inputs;
80  for(int i = 0; i < Ninput; i++)
81  inputs.push_back(m_InputStates[i].GetFourVector());
82 
83  // boost input vectors to CM frame
84  TLorentzVector TOT(0.,0.,0.,0.);
85  for(int i = 0; i < Ninput; i++) TOT += inputs[i];
86  TVector3 boost = TOT.BoostVector();
87  for(int i = 0; i < Ninput; i++) inputs[i].Boost(-boost);
88 
89  int ip_max[2];
90  int jp_max[2];
91  for(int i = 0; i < 2; i++) ip_max[i] = -1;
92  for(int i = 0; i < 2; i++) jp_max[i] = -1;
93  double metric_max = -1.;
94  // Loop over all 2-jet seed probes
95  int ip[2], jp[2];
96  for(ip[0] = 0; ip[0] < Ninput-1; ip[0]++){
97  for(ip[1] = ip[0]+1; ip[1] < Ninput; ip[1]++){
98  TVector3 nRef = inputs[ip[0]].Vect().Cross(inputs[ip[1]].Vect());
99  int Nhem[2];
100  TLorentzVector hem[2];
101  for(int i = 0; i < 2; i++){
102  Nhem[i] = 0;
103  hem[i].SetPxPyPzE(0.,0.,0.,0.);
104  }
105  // Loop over all jets
106  for(int i = 0; i < Ninput; i++){
107  if((i == ip[0]) || (i ==ip[1])) continue;
108  int ihem = int(inputs[i].Vect().Dot(nRef) > 0.);
109  Nhem[ihem]++;
110  hem[ihem] += inputs[i];
111  }
112  // assign 2 probes
113  for(jp[0] = 0; jp[0] < 2; jp[0]++){
114  for(jp[1] = 0; jp[1] < 2; jp[1]++){
115  if(jp[0] == jp[1] && Nhem[!jp[0]] == 0) continue;
116  TLorentzVector hem_probes[2];
117  for(int i = 0; i < 2; i++) hem_probes[i] = hem[i];
118  for(int i = 0; i < 2; i++) hem_probes[jp[i]] += inputs[ip[i]];
119  double metric = hem_probes[0].P() + hem_probes[1].P();
120  if(metric > metric_max){
121  metric_max = metric;
122  for(int i = 0; i < 2; i++) ip_max[i] = ip[i];
123  for(int i = 0; i < 2; i++) jp_max[i] = jp[i];
124  }
125  }
126  }
127  }
128  }
129  if(metric_max < 0)
130  return false;
131 
132  // initialize output states
133  for(int i = 0; i < 2; i++) GetChildState(i).ClearElements();
134  for(int i = 0; i < 2; i++) GetChildState(jp_max[i]).AddElement(m_InputStates[ip_max[i]]);
135  TVector3 nRef = inputs[ip_max[0]].Vect().Cross(inputs[ip_max[1]].Vect());
136  for(int i = 0; i < Ninput; i++){
137  if((i == ip_max[0]) || (i == ip_max[1])) continue;
138  int ihem = int(inputs[i].Vect().Dot(nRef) > 0.);
139  GetChildState(ihem).AddElement(m_InputStates[i]);
140  }
141  if(GetChildState(1).GetFourVector().M() > GetChildState(1).GetFourVector().M()){
142  vector<RFList<VisibleState> > flip;
143  for(int i = 0; i < 2; i++) flip.push_back(GetChildState(i).GetElements());
144  for(int i = 0; i < 2; i++) GetChildState(i).ClearElements();
145  for(int i = 0; i < 2; i++) GetChildState(i).AddElements(flip[!i]);
146  }
147 
148  // Execute depedancy Jigsaws
149  ExecuteDependancyJigsaws();
150 
151  return SetSpirit(true);
152  }

154  double MinMassesCombJigsaw::EvaluateMetric() const {
155  int N = GetNChildren();
156  vector<TLorentzVector> P;
157  for(int i = 0; i < N; i++)
158  P.push_back(GetChildState(i).GetFourVector());
159  N = m_DependancyStates.size();
160  for(int i = 0; i < N; i++)
161  P[i] += GetDependancyStates(i).GetFourVector();
162 
163  double prob = 1.;
164  TLorentzVector SUM(0.,0.,0.,0.);
165  for(int i = 1; i < N; i++)
166  SUM += P[i];
167  for(int i = 0; i < N-1; i++){
168  prob *= GetProb((P[i]+SUM).M(), P[i].M(), SUM.M());
169  SUM -= P[i+1];
170  }
171  return prob;
172  }
