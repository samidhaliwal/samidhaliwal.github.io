98  string RFLog::GetFormattedMessage(const string& message) {
99  string output = "";
100  int N = message.size();
101  double OFF = 18;
102  if(N-OFF > m_NMAX){
103  int Ncut = (N-OFF)/m_NMAX;
104  string::size_type previous_pos = 0;
105  for(int i = 0; i <= Ncut; i++){
106  int off = m_NMAX;
107  if(i == 0) off += OFF;
108  string line = message.substr(previous_pos, off);
109  if(i > 0){
110  if(m_Color)
111  output += m_ColorMap[m_CurType]+"<...>\x1b[0m ...";
112  else
113  output += "<...> ...";
114  }
115  output += line;
116  previous_pos += off;
117  if(previous_pos != N && i != Ncut) output += "...\n";
118  }
119  } else {
120  output = message;
121  }
122  return output;
123  }

125  void RFLog::Send(){
126  string source_name = GetFormattedSource();
127  string message = m_Message.str();
128  string::size_type previous_pos = 0, current_pos = 0;
129  if(m_PrintMap[m_CurType] && m_Ostr){
130  string prefix;
131  if(m_Color)
132  prefix = m_ColorMap[m_CurType]+"<"+m_TypeMap[m_CurType]+">";
133  else
134  prefix = "<"+m_TypeMap[m_CurType]+">";
135  for(int i = 0; i < 8-m_TypeMap[m_CurType].size(); i++){
136  prefix += ' ';
137  }
138  prefix += source_name+": ";
139  if(m_Color)
140  prefix +="\x1b[0m";
141  while (true) {
142  current_pos = message.find( '\n', previous_pos );
143  string line = message.substr( previous_pos, current_pos - previous_pos );
144 
145  ostringstream message_to_send;
146  message_to_send.setf(std::ios::adjustfield, std::ios::left);
147  line = GetFormattedMessage(prefix+line);
148  message_to_send << line << endl;
149 
150  *m_Ostr << message_to_send.str();
151  m_Ostr->flush();
152 
153  if (current_pos == message.npos) break;
154  previous_pos = current_pos + 1;
155  }
156  }
157 
158  if (m_CurType == LogError)
159  throw RestFramesException(m_Message.str());
160 
161  m_Message.str("");
162  m_CurType = m_DefType;
163  return;
164  }
