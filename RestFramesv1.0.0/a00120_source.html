101  bool LabRecoFrame::InitializeTreeStates(){
102  m_Log << LogVerbose << "Initializing States for analysis..." << m_End;
103 
104  m_TreeStates.Clear();
105  int Ng = m_TreeGroups.GetN();
106 
107  RFList<ReconstructionFrame> frames = GetListVisibleFrames();
108  int Nf = frames.GetN();
109  for(int f = 0; f < Nf; f++){
110  bool has_group = false;
111  for(int g = 0; g < Ng; g++){
112  if(m_TreeGroups[g].ContainsFrame(frames[f])){
113  has_group = true;
114  break;
115  }
116  }
117  if(!has_group) {
118  VisibleState& state = GetNewVisibleState();
119  state.AddFrame(frames[f]);
120  m_TreeStates += state;
121  }
122  }
123 
124  frames = GetListInvisibleFrames();
125  Nf = frames.GetN();
126  for(int f = 0; f < Nf; f++){
127  bool has_group = false;
128  for(int g = 0; g < Ng; g++){
129  if(m_TreeGroups[g].ContainsFrame(frames[f])){
130  has_group = true;
131  break;
132  }
133  }
134  if(!has_group){
135  m_Log << LogWarning;
136  m_Log << "Found InvisibleRecoFrame without an assigned group: " << endl;
137  m_Log << Log(frames[f]) << m_End;
138  return false;
139  }
140  }
141  m_Log << LogVerbose << "...Done initializing States" << m_End;
142  return true;
