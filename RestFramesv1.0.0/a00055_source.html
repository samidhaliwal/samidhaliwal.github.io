49 void example_04_HtoWWto4l(const string& output_name = "output_04.root"){

52  SetLogPrint(LogVerbose,true);

56  double mH = 1000.;
57  double wH = 0.04;
58  double mW = 80.;
59  double wW = 2.5;
60  double mL = 0.501;
61  double mN = 0.;
62  int Ngen = 100000;
63 
64  //
65  // Set up toy generation tree (not needed for reconstruction)
66  g_Log << LogInfo << "Initializing generator frames and tree" << g_End;
67  LabGenFrame LAB_G("LAB_G","LAB");
68  ResonanceGenFrame H_G("H_G","H");
69  ResonanceGenFrame Wa_G("Wa_G","W_{a}");
70  ResonanceGenFrame Wb_G("Wb_G","W_{b}");
71  VisibleGenFrame La_G("La_G","#it{l}_{a}");
72  InvisibleGenFrame Na_G("Na_G","#nu_{a}");
73  VisibleGenFrame Lb_G("Lb_G","#it{l}_{b}");
74  InvisibleGenFrame Nb_G("Nb_G","#nu_{b}");
75  LAB_G.SetChildFrame(H_G);
76  H_G.AddChildFrame(Wa_G);
77  H_G.AddChildFrame(Wb_G);
78  Wa_G.AddChildFrame(La_G);
79  Wa_G.AddChildFrame(Na_G);
80  Wb_G.AddChildFrame(Lb_G);
81  Wb_G.AddChildFrame(Nb_G);
82 
83  if(LAB_G.InitializeTree()){
84  g_Log << LogInfo;
85  g_Log << "Successfully initialized tree from LabFrame ";
86  g_Log << LAB_G.GetName() << endl;
87  g_Log << "Ready to set masses etc." << g_End;
88  } else
89  g_Log << LogError << "Unable to initialize tree from LabFrame: " << Log(LAB_G) << g_End;
90  // set Higgs masses
91  H_G.SetMass(mH);
92  H_G.SetWidth(wH);
93  // set W masses
94  Wa_G.SetMass(mW);
95  Wb_G.SetMass(mW);
96  Wa_G.SetWidth(wW);
97  Wb_G.SetWidth(wW);
98  // set lepton masses
99  La_G.SetMass(mL);
100  Lb_G.SetMass(mL);
101  // set neutrino masses
102  Na_G.SetMass(mN);
103  Nb_G.SetMass(mN);
104 
105  if(LAB_G.InitializeAnalysis()){
106  g_Log << LogInfo;
107  g_Log << "Successfully initialized analysis from LabFrame ";
108  g_Log << LAB_G.GetName() << endl;
109  g_Log << "Ready for event generation" << g_End;
110  } else
111  g_Log << LogError << "Unable to initialize analysis from LabFrame: " << Log(LAB_G) << g_End;
112 
113 
114  // Set up reco analysis tree
115  g_Log << LogInfo << "Initializing reconstruction frames and tree" << g_End;
116  LabRecoFrame LAB_R("LAB_R","LAB");
117  DecayRecoFrame H_R("H_R","H");
118  DecayRecoFrame Wa_R("Wa_R","W_{a}");
119  DecayRecoFrame Wb_R("Wb_R","W_{b}");
120  VisibleRecoFrame La_R("La_R","#it{l}_{a}");
121  InvisibleRecoFrame Na_R("Na_R","#nu_{a}");
122  VisibleRecoFrame Lb_R("Lb_R","#it{l}_{b}");
123  InvisibleRecoFrame Nb_R("Nb_R","#nu_{b}");
124  LAB_R.SetChildFrame(H_R);
125  H_R.AddChildFrame(Wa_R);
126  H_R.AddChildFrame(Wb_R);
127  Wa_R.AddChildFrame(La_R);
128  Wa_R.AddChildFrame(Na_R);
129  Wb_R.AddChildFrame(Lb_R);
130  Wb_R.AddChildFrame(Nb_R);
131 
132  if(LAB_R.InitializeTree()){
133  g_Log << LogInfo;
134  g_Log << "Successfully initialized tree from LabFrame ";
135  g_Log << LAB_R.GetName() << endl;
136  g_Log << "Ready for Group and Jigsaw initialization" << g_End;
137  } else
138  g_Log << LogError << "Unable to initialize tree from LabFrame: " << Log(LAB_R) << g_End;
139 
140  // define groups for the reconstruction trees
141  InvisibleGroup INV_R("INV_R","WIMP Jigsaws");
142  INV_R.AddFrame(Na_R);
143  INV_R.AddFrame(Nb_R);
144 
145  // define jigsaws for the reconstruction tree
146  SetMassInvJigsaw MinMassJigsaw_R("MINMASS_R", "Invisible system mass Jigsaw");
147  INV_R.AddJigsaw(MinMassJigsaw_R);
148 
149  SetRapidityInvJigsaw RapidityJigsaw_R("RAPIDITY_R", "Invisible system rapidity Jigsaw");
150  INV_R.AddJigsaw(RapidityJigsaw_R);
151  RapidityJigsaw_R.AddVisibleFrames((LAB_R.GetListVisibleFrames()));
152 
153  ContraBoostInvJigsaw ContraBoostJigsaw_R("CONTRA_R","Contraboost invariant Jigsaw");
154  INV_R.AddJigsaw(ContraBoostJigsaw_R);
155  ContraBoostJigsaw_R.AddVisibleFrames((Wa_R.GetListVisibleFrames()), 0);
156  ContraBoostJigsaw_R.AddVisibleFrames((Wb_R.GetListVisibleFrames()), 1);
157  ContraBoostJigsaw_R.AddInvisibleFrames((Wa_R.GetListInvisibleFrames()), 0);
158  ContraBoostJigsaw_R.AddInvisibleFrames((Wb_R.GetListInvisibleFrames()), 1);
159 
160  g_Log << LogInfo << "Initializing tree for analysis" << m_End;
161  // check reconstruction trees
162  if(LAB_R.InitializeAnalysis()){
163  g_Log << LogInfo;
164  g_Log << "...Successfully initialized tree for analysis from LabFrame ";
165  g_Log << LAB_R.GetName() << endl;
166  g_Log << "Ready event analysis" << g_End;
167  } else
168  g_Log << LogError << "...Unable to initialize analysis from LabFrame: " << Log(LAB_R) << g_End;
169 
170  DecayRecoFrame *W[2];
171  VisibleRecoFrame *L[2];
172  InvisibleRecoFrame *N[2];
173  W[0] = &Wa_R;
174  W[1] = &Wb_R;
175  L[0] = &La_R;
176  L[1] = &Lb_R;
177  N[0] = &Na_R;
178  N[1] = &Nb_R;
179 
180  // Now we book some histograms of kinematic variables
181  TH1D* h_MH = new TH1D("h_MH","h_MH",100,0.,2.);
182  TH1D* h_mH = new TH1D("h_mH","h_mH",100,0.,mH*2.);
183  TH1D* h_MW = new TH1D("h_MW","h_MW",100,0.,2.);
184  TH2D* h_MH_v_MW = new TH2D("h_MH_v_MW","h_MH_v_MW",50,0.,2.,50,0.,2.);
185  TH2D* h_mW_v_mW = new TH2D("h_mW_v_mW","h_mW_v_mW",50,0.,mH,50,0.,mH);
186 
187  for(int igen = 0; igen < Ngen; igen++){
188  if(igen%(Ngen/10) == 0)
189  g_Log << LogInfo << "Generating event " << igen << " of " << Ngen << g_End;
190 
191  // generate event
192  LAB_G.ClearEvent(); // clear the gen tree
193  double PTH = mH*gRandom->Rndm();
194  LAB_G.SetTransverseMomenta(PTH); // give the Higgs some Pt
195  double PzH = mH*(2.*gRandom->Rndm()-1.);
196  LAB_G.SetLongitudinalMomenta(PzH); // give the Higgs some Pz
197  LAB_G.AnalyzeEvent(); // generate a new event
198 
199  // analyze event
200  TVector3 MET = LAB_G.GetInvisibleMomentum(); // Get the MET from gen tree
201  MET.SetZ(0.);
202 
203  // give the signal-like tree the event info and analyze
204  LAB_R.ClearEvent(); // clear the signal-like tree
205  INV_R.SetLabFrameThreeVector(MET); // Set the MET in reco tree
206  La_R.SetLabFrameFourVector(La_G.GetFourVector());
207  Lb_R.SetLabFrameFourVector(Lb_G.GetFourVector());
208  LAB_R.AnalyzeEvent(); // analyze the event
209 
211  // Observable Calculations
213 
214  //
215  // signal tree observables
216  //
217 
218  //*** Higgs mass
219  double MH = H_R.GetMass();
220  double MW = Wa_R.GetMass();
221 
222  h_MH->Fill(MH/H_G.GetMass());
223  h_mH->Fill(H_G.GetMass());
224  h_MW->Fill(MW/mW);
225  h_MH_v_MW->Fill(MH/H_G.GetMass(),MW/mW);
226  h_mW_v_mW->Fill(Wa_G.GetMass(),Wb_G.GetMass());
227  }
228 
229  setstyle();
230  string plot_title = "H #rightarrow W(#it{l} #nu)W(#it{l} #nu)";
231  TCanvas *c_MH = Plot_Me("c_MH", h_MH, "M_{H} / m_{H}^{true}", plot_title);
232  TCanvas *c_mH = Plot_Me("c_mH", h_mH, "m_{H}^{true} [GeV]", plot_title);
233  TCanvas *c_MW = Plot_Me("c_MW", h_MW, "M_{W} / m_{W}^{true}", plot_title);
234  TCanvas *c_MH_v_MW = Plot_Me("c_MH_v_MW", h_MH_v_MW, "M_{H} / m_{H}^{true}", "M_{W} / m_{W}^{true}", plot_title);
235  TCanvas *c_mW_v_mW = Plot_Me("c_mW_v_mW", h_mW_v_mW, "m_{Wa}^{true} [GeV]", "m_{Wb}^{true} [GeV]", plot_title);
236 
237  TreePlot* tree_plot = new TreePlot("TreePlot","TreePlot");
238 
239  // generator tree
240  tree_plot->SetFrameTree(LAB_G);
241  tree_plot->Draw("GenTree", "Generator Tree");
242 
243  // reco tree
244  tree_plot->SetFrameTree(LAB_R);
245  tree_plot->Draw("RecoTree", "Reconstruction Tree");
246 
247  // reco tree
248  tree_plot->SetGroupTree(INV_R);
249  tree_plot->Draw("InvTree", "Invisible Jigsaws");

251 
252 
253  g_Log << LogInfo << "Finished" << g_End;
